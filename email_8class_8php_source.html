64 $mail->SMTPSecure = PHPMailer::ENCRYPTION_STARTTLS;

66 $mail->SMTPSecure = PHPMailer::ENCRYPTION_SMTPS;
67 }
68
69 // Configuration de l'expéditeur

72 $this->mailConfig['from_name']
73 );
