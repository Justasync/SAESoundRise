10/**
11 * @brief Bascule l'état "aimé" d'une chanson
12 *
13 * @details Cette fonction est appelée lorsqu'un utilisateur clique sur le bouton
14 * de like d'une chanson. Elle envoie une requête POST au serveur pour
15 * basculer l'état du like et met à jour l'icône en fonction de la réponse.
16 *
17 * @param {Event} event - L'événement de clic déclenché
18 * @param {number|string} chansonId - L'identifiant unique de la chanson
19 *
20 * @returns {void}
21 *
22 * @example
23 * // Utilisation dans le HTML :
24 * // <button onclick="toggleLikeInline(event, 123)">Like</button>
25 */
26window.toggleLikeInline = function(event, chansonId) {
27 // Empêche la propagation de l'événement aux éléments parents
28 event.stopPropagation();
29 console.log('toggleLikeInline appelé pour', chansonId);
30
31 // Récupération du bouton et de l'icône associés
32 const btn = (event.currentTarget) ? event.currentTarget : event.target.closest('.like-btn');
33 const icon = btn ? btn.querySelector('i') : null;
34
35 // Envoi de la requête AJAX au serveur
36 fetch('/?controller=chanson&method=toggleLike', {
37 method: 'POST',
38 headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
39 body: `idChanson=${encodeURIComponent(chansonId)}`
40 })
41 .then(async (res) => {
42 const text = await res.text();
43 try {
44 const data = JSON.parse(text);
45 return { ok: res.ok, status: res.status, data };
46 } catch (err) {
47 console.error('Réponse non JSON du serveur pour toggleLike:', text);
48 throw err;
49 }
50 })
51 .then(({ status, data }) => {
52 console.log('Réponse toggleLikeInline', status, data);
53
54 // Mise à jour de l'icône selon l'état du like
55 if (data.liked) {
56 // Chanson aimée : affiche un cœur plein rouge
57 if (icon) {
58 icon.classList.remove('bi-heart');
59 icon.classList.add('bi-heart-fill');
60 icon.classList.add('text-danger');
61 }
62 } else {
63 // Chanson non aimée : affiche un cœur vide
64 if (icon) {
65 icon.classList.remove('bi-heart-fill');
66 icon.classList.add('bi-heart');
67 icon.classList.remove('text-danger');
68 }
69 }
70 })
71 .catch(err => console.error('Erreur toggleLikeInline:', err));
72};
