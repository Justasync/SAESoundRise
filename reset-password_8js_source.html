46 /**
47 * Vérifie si le mot de passe respecte toutes les exigences.
48 * @param {string} password - Le mot de passe à vérifier.
49 * @returns {Object} Objet contenant l'état de chaque exigence.
50 */
51 function checkPasswordRequirements(password) {
52 return {
53 length: password.length >= 8,
54 uppercase: /[A-Z]/.test(password),
55 number: /[0-9]/.test(password),
56 special: /[!@#$%^&*()_+\-=\[\]{}|;:'",.<>?/~`]/.test(password)
57 };
58 }

60 /**
61 * Calcule la force du mot de passe.
62 * @param {Object} requirements - L'objet des exigences.
63 * @returns {Object} Objet avec classe et pourcentage.
64 */
65 function getPasswordStrength(requirements) {
66 const count = Object.values(requirements).filter(Boolean).length;
67 if (count <= 1) {
68 return { class: 'bg-danger', width: '33%' };
69 }
70 if (count <= 3) {
71 return { class: 'bg-warning', width: '66%' };
72 }
73 return { class: 'bg-success', width: '100%' };
74 }
